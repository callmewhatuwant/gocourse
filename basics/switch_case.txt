// interface for all datatypes
func checkType(x interface{})  {
	switch x.(type){
	case int:
		fmt.Println("int")
	case float64:
	fmt.Println("float")
	case string:
		fmt.Println("string")
	default:
		fmt.Println("Unkown")
	}
	
}
